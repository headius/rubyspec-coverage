package org.jruby.embed.osgi.internal;

import org.jruby.embed.osgi.utils.OSGiBundleClassLoaderHelper;
import org.osgi.framework.Bundle;

/**
 * @author awj
 *
 */
public class ReflectiveOSGiClassLoaderAdapter implements IOSGiClassLoaderAdapter {

    /**
     * Delegate to the old OSGiBundleClassLoaderHelper
     * @see org.jruby.embed.osgi.internal.IOSGiClassLoaderAdapter#getClassLoader(org.osgi.framework.Bundle)
     */
    @Override
    public ClassLoader getClassLoader(Bundle bundle) {
        return OSGiBundleClassLoaderHelper.getBundleClassLoader(bundle);
    }
}
