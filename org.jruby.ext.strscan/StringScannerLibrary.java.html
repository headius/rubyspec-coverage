package org.jruby.ext.strscan;

/**
 * @author kscott
 *
 */
public class StringScannerLibrary implements Library {

	/**
	 * @see org.jruby.runtime.load.Library#load(org.jruby.Ruby)
	 */

		RubyStringScanner.createScannerClass(runtime);
	}

}
